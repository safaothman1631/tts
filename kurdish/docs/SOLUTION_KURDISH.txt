🌐 بە ئینتەرنێت (باشترینە):
─────────────────────────────

from gtts import gTTS
import os

# دەقی کوردی
text = "سڵاو! چۆنی براکەم؟"

# دروستکردنی دەنگ
print(f"🎤 {text}")
tts = gTTS(text=text, lang='ar')
tts.save('output/voice.mp3')

# لێدانی دەنگ
os.system('start output/voice.mp3')

print("✅ دەنگ لێدرا!")


⚡ بێ ئینتەرنێت:
────────────────

import pyttsx3

# دروستکردنی بزوێنەر
engine = pyttsx3.init()

# دەق بە لاتینی (وەک کوردی دەبێت)
texts = [
    "sallaw",           # سڵاو
    "chonee",           # چۆنی
    "bayanee bash",     # بەیانی باش
]

for text in texts:
    print(f"🎤 {text}")
    engine.say(text)
    engine.runAndWait()

print("✅ تەواو!")
