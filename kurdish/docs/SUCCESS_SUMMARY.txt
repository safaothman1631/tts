┌───────────────────────────────────────────────────────────────────────┐
│ نموونەی کۆد / Code Example                                            │
└───────────────────────────────────────────────────────────────────────┘

from tts_pipeline import TTSPipeline

# دروستکردنی pipeline
pipeline = TTSPipeline(engine_type="gtts")

# ١. قسەکردنی دەقی کوردی
pipeline.process_and_speak("سڵاو! بەخێربێیت بۆ سیستەمەکە")

# ٢. پاشەکەوتکردن لە فایل
pipeline.process_and_save(
    "کوردستان خاکێکی جوانە",
    output_filename="my_kurdish_audio"
)

# ٣. چەند دەقێک لە یەکجار
texts = ["بەیانی باش", "ئێوارەتان باش", "شەوتان خۆش"]
pipeline.batch_process(texts, output_prefix="greetings")

🔧 بزوێنەری TTS: pyttsx3 (ئۆفلاین) + gTTS (ئۆنلاین)
🔧 پێشنیار بۆ کوردی: gTTS (engine_type="gtts")
🔧 ناسینەوەی زمان: langdetect (کوردی → "fa")
🔧 دەنگی کوردی: دەنگی عەرەبی لە gTTS
🔧 پێداویستییەکان: Python 3.8+, ئینتەرنێت بۆ gTTS
